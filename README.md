<script>
  // Texto que deseas mostrar
  
  var text = "Hola, soy Beltran Puma";

  // Tiempo de espera antes de comenzar a escribir
  
  var delay = 1000;

  // Tiempo entre cada letra (en milisegundos)
  
  var speed = 100;

  // Funci√≥n para animar el texto
  
  function typeWriter(text, i, fnCallback) {
    if (i < text.length) {
      document.getElementById("typed-text").innerHTML += text.charAt(i);
      setTimeout(function () {
        typeWriter(text, i + 1, fnCallback);
      }, speed);
    } else if (typeof fnCallback == "function") {
      setTimeout(fnCallback, delay);
    }
  }

  // Iniciar la animaci√≥n
  
  function startTextAnimation(i) {
    if (typeof text === "undefined") return;
    if (i < text.length) {
      typeWriter(text, 0, function () {
        startTextAnimation(i);
      });
    }
  }

  // Iniciar la animaci√≥n cuando la p√°gina est√© cargada
  
  document.addEventListener("DOMContentLoaded", function () {
    if (text) {
      startTextAnimation(0);
    }
  });
</script>
